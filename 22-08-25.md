Solve a Quadratic Equation

import cmath
a, b, c = 1, -3, 2
d = cmath.sqrt(b**2 - 4*a*c)
root1 = (-b + d) / (2*a)
root2 = (-b - d) / (2*a)
print("Roots:", root1, root2)
