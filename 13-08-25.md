#AMSTRONG NUMBER
num = int(input())  
digits = [int(d) for d in str(num)]   
power = len(digits)                  
total = 0

for d in digits:                      
    total += d ** power               

if total == num:                      
    print("Armstrong number")
else:
    print("Not an Armstrong number")

#PERFECT NUMBER
n=int(input())
sum = 0
for i in range(1, n):
    if n%i==0:           
        sum += i   
if sum==n:
    print("Perfect Number")
else:
    print("NOT a Perfect Number")

#FIBONACCI SERIES
n = int(input())
a = 0
b = 1
print("Fibonacci Series:")
for i in range(n):
    print(a, end=" ") 
    c = a + b         
    a = b             
    b = c
