#EVEN OR ODD NUMBER
n=int(input())
if n%2==0:
    print("Even")
else:
    print("Odd")

#FACTORIAL NUMBER
n = int(input())
fact = 1
for i in range(2, n + 1):  
    fact *= i
print(fact)

#PRIME NUMBER
n=int(input())
if n>0:
    print("Prime")
else:
    for i in range(2, n):
        if n%i==0:
            print("Not Prime")
            break
    else:
        print("Prime")
